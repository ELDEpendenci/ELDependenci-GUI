001package com.ericlam.mc.eldgui.component;
002
003import org.bukkit.Material;
004
005/**
006 * 基礎組件工廠，所有組件工廠必須繼承此類
007 * @param <Factory> 組件工廠
008 */
009public interface ComponentFactory<Factory extends ComponentFactory<Factory>> {
010
011    /**
012     *
013     * @param material 圖示
014     * @return this
015     */
016    Factory icon(Material material);
017
018    /**
019     *
020     * @param amount 數量
021     * @return this
022     */
023    Factory number(int amount);
024
025    /**
026     * 綁定組件屬性
027     * @param key 鍵
028     * @param value 數值
029     * @return this
030     */
031    Factory bind(String key, Object value);
032
033    /**
034     * 創建組件
035     * @return this
036     */
037    Component create();
038
039}
