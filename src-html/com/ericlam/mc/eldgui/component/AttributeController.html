001package com.ericlam.mc.eldgui.component;
002
003import org.bukkit.inventory.ItemStack;
004
005import javax.annotation.Nullable;
006
007/**
008 * 物品屬性編輯器
009 */
010public interface AttributeController {
011
012    /**
013     * 預設綁定 Model 屬性數值 的 標識文字
014     */
015    String VALUE_TAG = "input-value";
016    /**
017     * 預設綁定 Model 屬性名稱 的 標識文字
018     */
019    String FIELD_TAG = "input-field";
020
021    /**
022     * 設置屬性
023     * @param itemStack 物品
024     * @param key 鍵
025     * @param value 數值
026     */
027    void setAttribute(ItemStack itemStack, String key, Object value);
028
029    /**
030     * 設置 該 pattern 内所有物品的屬性
031     * @param pattern pattern
032     * @param key 鍵
033     * @param value 數值
034     */
035    void setAttribute(char pattern, String key, Object value);
036
037    /**
038     * 獲取物品的屬性
039     * @param item 物品
040     * @param key 鍵
041     * @param <C> 獲取類型
042     * @return 數值
043     */
044    @Nullable
045    <C> C getAttribute(ItemStack item, String key);
046
047}
