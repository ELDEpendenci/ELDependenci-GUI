001package com.ericlam.mc.eldgui.component.modifier;
002
003import org.bukkit.entity.Player;
004import org.bukkit.event.player.PlayerEvent;
005
006/**
007 * 可透過監聽指定玩家事件修改組件屬性
008 * @param <E> 玩家事件
009 */
010public interface Listenable<E extends PlayerEvent> extends Activatable {
011
012    /**
013     * 開始監聽時的行爲
014     * @param player 玩家
015     */
016    void onListen(Player player);
017
018    /**
019     *
020     * @return 最長等待時間(ticks)
021     */
022    long getMaxWaitingTime();
023
024    /**
025     * 傳回事件後的動作
026     * @param event 指定事件
027     */
028    void callBack(E event);
029
030    /**
031     *
032     * @return 事件類型
033     */
034    Class<E> getEventClass();
035
036}
