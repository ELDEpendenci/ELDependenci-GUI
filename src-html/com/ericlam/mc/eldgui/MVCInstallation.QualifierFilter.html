001package com.ericlam.mc.eldgui;
002
003import com.ericlam.mc.eldgui.component.ComponentFactory;
004import com.ericlam.mc.eldgui.exception.ExceptionViewHandler;
005import com.ericlam.mc.eldgui.view.LoadingView;
006import org.bukkit.event.inventory.InventoryInteractEvent;
007
008import java.lang.annotation.Annotation;
009import java.util.List;
010
011/**
012 * ELDependenci-MVC 安裝器
013 */
014public interface MVCInstallation {
015
016    /**
017     * 安裝界面互動事件自定義過濾
018     *
019     * @param qualifier 指定的標注
020     * @param predicate 指定自定義過濾
021     * @param <A>       標注類
022     */
023    <A extends Annotation> void registerQualifier(Class<A> qualifier, QualifierFilter<A> predicate);
024
025    /**
026     * 安裝 controllers (控制器)
027     *
028     * @param controllers controllers
029     */
030    void registerControllers(Class<?>... controllers);
031
032    /**
033     * 新增自定義異常處理器
034     *
035     * @param exceptionHandlers 異常處理器
036     */
037    void addExceptionViewHandlers(List<Class<? extends ExceptionViewHandler>> exceptionHandlers);
038
039    /**
040     * 設置全局異常處理器
041     * @param exceptionHandler 異常處理器
042     */
043    void setGlobalExceptionHandler(Class<? extends ExceptionViewHandler> exceptionHandler);
044
045    /**
046     * 設置全局異步加載界面
047     * @param loadingView 異步加載界面
048     */
049    void setGlobalLoadingView(Class<? extends LoadingView> loadingView);
050
051    /**
052     * 新增自定義組件工廠
053     * @param factory 組件類，必須爲 interface
054     * @param implement 組件實作類
055     * @param <T> 組件工廠類
056     * @param <E> 組件工廠實作類
057     */
058    <T extends ComponentFactory<T>, E extends T> void addComponentFactory(Class<T> factory, Class<E> implement);
059
060    @FunctionalInterface
061    interface QualifierFilter<A extends Annotation> {
062        boolean checkIsPass(InventoryInteractEvent interactEvent, char pattern, A annotation);
063    }
064
065}
