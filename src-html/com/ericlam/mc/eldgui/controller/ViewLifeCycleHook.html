001package com.ericlam.mc.eldgui.controller;
002
003import com.ericlam.mc.eldgui.UISession;
004import com.ericlam.mc.eldgui.view.View;
005import org.bukkit.entity.Player;
006
007/**
008 * 界面生命周期挂鈎
009 */
010public interface ViewLifeCycleHook {
011
012    /**
013     * 新界面創建後
014     * @param player 玩家
015     * @param view 創建後的新界面
016     * @param session Session
017     */
018    void postUpdateView(Player player, Class<View<?>> view, UISession session);
019
020    /**
021     * 界面摧毀前(更新前)
022     * @param player 玩家
023     * @param view 舊界面
024     * @param session Session
025     */
026    void preViewDestroy(Player player, Class<View<?>> view, UISession session);
027
028}
