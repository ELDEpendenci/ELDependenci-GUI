001package com.ericlam.mc.eldgui.view;
002
003import com.ericlam.mc.eldgui.InventoryTemplate;
004
005import java.lang.annotation.ElementType;
006import java.lang.annotation.Retention;
007import java.lang.annotation.RetentionPolicy;
008import java.lang.annotation.Target;
009
010/**
011 * 在 {@link View} 上使用標注，用於指定 template 文件來進行預設渲染。
012 */
013@Retention(RetentionPolicy.RUNTIME)
014@Target(ElementType.TYPE)
015public @interface UseTemplate {
016
017    /**
018     *
019     * @return template 文件
020     */
021    String template();
022
023    /**
024     *
025     * @return 指定文件組類別
026     */
027    Class<? extends InventoryTemplate> groupResource();
028
029}
