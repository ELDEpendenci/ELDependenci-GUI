001package com.ericlam.mc.eldgui;
002
003import javax.annotation.Nullable;
004
005/**
006 * 玩家在打開界面到關閉界面時的數據容器，用於在各個 Controller 之間傳遞資料用
007 */
008public interface UISession {
009
010    /**
011     * 透過 key 獲取數據
012     * @param key 鍵
013     * @param <T> 數據類型
014     * @return 數據，可爲 null
015     */
016    @Nullable
017    <T> T getAttribute(String key);
018
019    /**
020     * 透過 key 提取數據並在 Session 中刪除
021     * @param key 鍵
022     * @param <T> 數據類型
023     * @return 數據，可爲 null
024     */
025    @Nullable
026    <T> T pollAttribute(String key);
027
028    /**
029     * 設置數據到 Session
030     * @param key 鍵
031     * @param value 數值
032     */
033    void setAttribute(String key, Object value);
034
035
036
037}
